analytical=read.csv("accounts_analytical.csv",header=T)
#Since we are focusing on the loan, we can drop the accounts when loan amount is NA.
aa1=analytical %>% filter(!is.na(loan_amount))
b<-ggplot(data=aa1,aes(x=loan_payment,y=loan_amount))
b+geom_point(aes(color=factor(loan_term)))+labs(title="Relationship between loan term and payment")+theme(plot.title=element_text(hjust=0.5))

cat("This is quite obvious that loan payment has a linear relationship with the loan amount. The loan payment for the same loan amount will result in differences: the longer the loan_term, the smaller the payment.")
## This is quite obvious that loan payment has a linear relationship with the loan amount. The loan payment for the same loan amount will result in differences: the longer the loan_term, the smaller the payment.
c<-ggplot(data=aa1,aes(x=loan_status,fill=loan_default))
c+geom_bar(position=position_dodge())+labs(title="Comparison of loan default and loan status")+theme(plot.title=element_text(hjust=0.5))

cat("Most of loans are not default, and the defaulted loans has smaller percentage in current loans than expired loans.")
## Most of loans are not default, and the defaulted loans has smaller percentage in current loans than expired loans.
